**Create `src/routes/org/[slug]/usage/budget/+server.ts`:**

API endpoint for saving budgets. Handles POST requests for both per-member and org-default budgets.

```typescript
import type { RequestHandler } from './$types';
import { json, error } from '@sveltejs/kit';
import { db } from '$lib/server/db';
import { appBudgets } from '$lib/server/db/schema';
import { eq, and, isNull } from 'drizzle-orm';

export const POST: RequestHandler = async ({ request, locals, params }) => {
  if (!locals.user) return error(401, 'Unauthorized');

  // Check admin/owner role (from layout data)
  // ... role check logic using appOrgMembers

  const body = await request.json();
  const { userId, hardLimit, softLimit, resetDay, isOrgDefault } = body;

  // Validate: hardLimit >= softLimit if both set, min $1, resetDay 1-28
  if (hardLimit !== null && softLimit !== null && hardLimit < softLimit) {
    return json({ error: 'Hard limit must be >= soft limit' }, { status: 400 });
  }
  if (resetDay !== undefined && (resetDay < 1 || resetDay > 28)) {
    return json({ error: 'Reset day must be between 1 and 28' }, { status: 400 });
  }

  // Convert dollar amounts to cents (multiply by 100)
  const hardLimitCents = hardLimit !== null && hardLimit !== undefined ? Math.round(hardLimit * 100) : null;
  const softLimitCents = softLimit !== null && softLimit !== undefined ? Math.round(softLimit * 100) : null;

  // Look up org by slug
  // ... get orgId from params.slug

  // Upsert: find existing budget for this user/org (or org default)
  const existing = await db.select().from(appBudgets).where(
    isOrgDefault
      ? and(eq(appBudgets.orgId, orgId), eq(appBudgets.isOrgDefault, true))
      : and(eq(appBudgets.orgId, orgId), eq(appBudgets.userId, userId))
  ).limit(1);

  if (existing.length > 0) {
    await db.update(appBudgets)
      .set({
        hardLimitCents,
        softLimitCents,
        resetDay: resetDay ?? existing[0].resetDay,
        updatedAt: new Date()
      })
      .where(eq(appBudgets.id, existing[0].id));
  } else {
    await db.insert(appBudgets).values({
      id: crypto.randomUUID(),
      orgId,
      userId: isOrgDefault ? null : userId,
      hardLimitCents,
      softLimitCents,
      resetDay: resetDay ?? 1,
      isOrgDefault: isOrgDefault ?? false
    });
  }

  return json({ success: true });
};

// DELETE handler for removing a budget
export const DELETE: RequestHandler = async ({ request, locals, params }) => {
  // Similar auth check, then delete the budget record
  const body = await request.json();
  const { userId } = body;
  // ... delete from appBudgets where orgId and userId match
  return json({ success: true });
};
```

The endpoint must:
- Verify the requesting user is admin/owner of the org (query appOrgMembers)
- Accept `userId` (string, for per-member) or `isOrgDefault: true` (for org default)
- Accept `hardLimit` and `softLimit` as dollar amounts (convert to cents for storage)
- Accept optional `resetDay` (1-28, only used for org default)
- Upsert logic: update if exists, insert if new
- DELETE removes the budget record entirely

```typescript
// Check if user has a budget and is approaching limit
import { appBudgets, appUsageLogs } from '$lib/server/db/schema';

// Find user's budget (specific or org default)
const budgets = await db.select().from(appBudgets)
  .where(and(
    eq(appBudgets.orgId, currentOrg.id),
    or(
      eq(appBudgets.userId, locals.user.id),
      and(isNull(appBudgets.userId), eq(appBudgets.isOrgDefault, true))
    )
  ));

const budget = budgets.find(b => b.userId === locals.user.id) ?? budgets.find(b => b.isOrgDefault);

let budgetWarning = null;
if (budget && (budget.hardLimitCents || budget.softLimitCents)) {
  const resetDate = getBudgetResetDate(budget.resetDay);
  // Get current spend
  const spendResult = await db.select({
    total: sql<string>`COALESCE(SUM(CAST(${appUsageLogs.cost} AS numeric)), 0)`
  }).from(appUsageLogs).where(and(
    eq(appUsageLogs.orgId, currentOrg.id),
    eq(appUsageLogs.userId, locals.user.id),
    gte(appUsageLogs.createdAt, resetDate)
  ));

  const currentSpendDollars = parseFloat(spendResult[0]?.total ?? '0');
  const limitCents = budget.hardLimitCents ?? budget.softLimitCents;
  const limitDollars = limitCents ? limitCents / 100 : null;

  if (limitDollars && currentSpendDollars >= limitDollars * 0.9) {
    budgetWarning = { currentSpend: currentSpendDollars, limit: limitDollars };
  }
}
```
